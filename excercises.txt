function vowelsAndConsonants(s) {
 let vowels_s = [];
 let consonants_s = [];

 const vowels=['a','e','i','o','u'];

 for(let i = 0; i < s.length; i++ )    
 {
     let c = s.charAt(i)
     if(vowels.indexOf(c) !==-1)
     {
        vowels_s.push(c);
     }
     else{
         consonants_s.push(c);
     }
 }
 for(let i = 0; i< vowels_s.length; i++){
    console.log(vowels_s[i]);
 }
 for(let i = 0; i< consonants_s.length; i++){
    console.log(consonants_s[i]);
 }

// Complete the compareTriplets function below.
function compareTriplets(a, b) {
    let res = [0,0];
    for (let i = 0 ; i<3 ; i++){
        if (a[i] > b[i]){
            res[0]+=1
        }else if (b[i] > a[i]){
            res[1]+=1
        }
    }
    return res;
}

function getCount(objects) {
   return objects.reduce((accum,val)=>{
       const {x, y} = val;
       return (x == y) ? accum + 1: accum;
   },0);

class Polygon {
    constructor(sides){
        this.sides = sides;
    }

    perimeter(){
        return this.sides.reduce((acc,val)=> (val+acc), 0)
    }
}
